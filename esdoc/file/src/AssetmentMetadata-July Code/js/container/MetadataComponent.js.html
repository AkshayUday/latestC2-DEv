const validate = values => {
  const errors = {"PMD":{}, "MVMData":{}, "PLT":{}, "QLT":{}}
  if (!values.PMD.Author) {
    errors.PMD.Author = 'Required'
  } else if (values.PMD.Author.length > 5) {
    errors.PMD.Author = 'Must be 5 characters or less'
  }

  if (!values.MVMData.AssignmentTitle) {
    errors.MVMData.AssignmentTitle = 'Required'
  } else if (values.MVMData.AssignmentTitle.length > 10) {
    errors.MVMData.AssignmentTitle = 'Must be 10 characters or less'
  }

  if (!values.QLT.tags || values.QLT.tags.length===0) {
    errors.QLT.tags = 'Required'
  } 

  if (!values.PLT.tags || values.PLT.tags.length===0) {
    errors.PLT.tags = 'Required'
  }
  return errors
}
